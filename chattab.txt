   328	function ChatTab() {
   329	  const [messages, setMessages] = useState<any[]>([]);
   330	  const [input, setInput] = useState("");
   331	  const [isLoading, setIsLoading] = useState(false);
   332	  const [selectedFile, setSelectedFile] = useState<{name: string, type: string, base64: string} | null>(null);
   333	  const [isDragging, setIsDragging] = useState(false);
   334	  const fileInputRef = useRef<HTMLInputElement>(null);
   335	  const messagesEndRef = useRef<HTMLDivElement>(null);
   336	
   337	  useEffect(() => {
   338	    messagesEndRef.current?.scrollIntoView({ behavior: "smooth" });
   339	  }, [messages]);
   340	
   341	  const handleFile = (file: File) => {
   342	    const reader = new FileReader();
   343	    reader.onload = (event) => {
   344	      const base64String = (event.target?.result as string).split(',')[1];
   345	      setSelectedFile({
   346	        name: file.name,
   347	        type: file.type,
   348	        base64: base64String
   349	      });
   350	    };
   351	    reader.readAsDataURL(file);
   352	    
   353	    if (fileInputRef.current) {
   354	       fileInputRef.current.value = "";
   355	    }
   356	  };
   357	
   358	  const handleFileChange = (e: React.ChangeEvent<HTMLInputElement>) => {
   359	    const file = e.target.files?.[0];
   360	    if (!file) return;
   361	    handleFile(file);
   362	  };
   363	
   364	  const handleDragOver = (e: React.DragEvent) => {
   365	    e.preventDefault();
   366	    setIsDragging(true);
   367	  };
   368	
   369	  const handleDragLeave = (e: React.DragEvent) => {
   370	    e.preventDefault();
   371	    setIsDragging(false);
   372	  };
   373	
   374	  const handleDrop = (e: React.DragEvent) => {
   375	    e.preventDefault();
   376	    setIsDragging(false);
   377	    const file = e.dataTransfer.files?.[0];
   378	    if (!file) return;
   379	    handleFile(file);
   380	  };
   381	
   382	  const removeFile = () => {
   383	    setSelectedFile(null);
   384	  };
   385	
   386	  const handleSend = async () => {
   387	    if ((!input.trim() && !selectedFile) || isLoading) return;
   388	
   389	    const newUserMessageParts: any[] = [];
   390	    if (input.trim()) {
   391	      newUserMessageParts.push({ text: input.trim() });
   392	    }
   393	    
   394	    if (selectedFile) {
   395	      newUserMessageParts.push({
   396	        inlineData: {
   397	          data: selectedFile.base64,
   398	          mimeType: selectedFile.type
   399	        }
   400	      });
   401	    }
   402	
   403	    const newUserMsg = { role: "user", parts: newUserMessageParts, originalFile: selectedFile };
   404	    const newMessages = [...messages, newUserMsg];
   405	    
   406	    setMessages(newMessages);
   407	    setInput("");
   408	    setSelectedFile(null);
   409	    setIsLoading(true);
   410	
   411	    try {
   412	      const response = await fetch("/api/query/lab-brain", {
   413	        method: "POST",
   414	        headers: { "Content-Type": "application/json" },
   415	        // send history without the `originalFile` custom field
   416	        body: JSON.stringify({
   417	          messages: newMessages.map(m => ({
   418	            role: m.role,
   419	            parts: m.parts
   420	          }))
   421	        }),
   422	      });
   423	      const data = await response.json();
   424	      
   425	      if (response.ok) {
   426	        setMessages(prev => [...prev, { role: "model", parts: [{ text: data.text }] }]);
   427	      } else {
   428	        setMessages(prev => [...prev, { role: "model", parts: [{ text: `Error: ${data.error}` }] }]);
   429	      }
   430	    } catch (error) {
   431	       setMessages(prev => [...prev, { role: "model", parts: [{ text: "Failed to communicate with the server." }] }]);
   432	    } finally {
   433	      setIsLoading(false);
   434	    }
   435	  };
   436	
   437	  return (
   438	    <div 
   439	      className={`flex flex-col h-[600px] bg-white dark:bg-slate-900/40 border ${isDragging ? 'border-indigo-500' : 'border-slate-200 dark:border-slate-800'} rounded-3xl overflow-hidden relative transition-colors duration-200`}
   440	      onDragOver={handleDragOver}
   441	      onDragLeave={handleDragLeave}
   442	      onDrop={handleDrop}
   443	    >
   444	      {isDragging && (
   445	        <div className="absolute inset-0 z-50 flex items-center justify-center bg-indigo-500/10 backdrop-blur-sm border-2 border-indigo-500 border-dashed rounded-3xl">
   446	          <div className="flex flex-col items-center gap-4 p-8 bg-white/80 dark:bg-slate-900/80 rounded-2xl shadow-xl backdrop-blur-md">
   447	            <div className="w-16 h-16 bg-indigo-100 dark:bg-indigo-900/50 rounded-full flex items-center justify-center text-indigo-600 dark:text-indigo-400">
   448	              <FileText size={32} />
   449	            </div>
   450	            <p className="text-lg font-bold text-slate-800 dark:text-slate-200">Drop document here</p>
   451	            <p className="text-sm text-slate-500">Supports Images, PDFs, and Text files</p>
   452	          </div>
   453	        </div>
   454	      )}
   455	
   456	      {/* Header */}
   457	      <div className="px-6 py-4 border-b border-slate-800 flex items-center justify-between bg-slate-900/50">
   458	        <div className="flex items-center gap-3">
   459	          <MessageSquare size={18} className="text-indigo-400" />
   460	          <h2 className="font-bold text-sm uppercase tracking-widest text-slate-800 dark:text-slate-200">Hermes Assistant</h2>
   461	        </div>
   462	        <div className="text-[10px] font-bold text-indigo-400 px-3 py-1 bg-indigo-500/10 border border-indigo-500/20 rounded-full uppercase tracking-wider">gemini-2.5-flash</div>
   463	      </div>
   464	
   465	      {/* Messages */}
   466	      <div className="flex-1 overflow-y-auto p-6 space-y-6">
   467	        {messages.length === 0 && (
   468	          <div className="h-full flex flex-col items-center justify-center text-slate-500 dark:text-slate-500 space-y-4">
   469	             <MessageSquare size={48} className="opacity-20" />
   470	             <p className="text-sm">Start a conversation or drag and drop a document to begin.</p>
   471	          </div>
   472	        )}
   473	        {messages.map((msg, idx) => (
   474	          <div key={idx} className={`flex ${msg.role === 'user' ? 'justify-end' : 'justify-start'}`}>
   475	            <div className={`max-w-[80%] rounded-2xl px-5 py-4 ${
   476	              msg.role === 'user' 
   477	                ? 'bg-indigo-600 text-white rounded-tr-sm' 
   478	                : 'bg-slate-50 dark:bg-slate-800/40 text-slate-800 dark:text-slate-200 rounded-tl-sm border border-slate-200 dark:border-slate-700/30'
   479	            }`}>
   480	              {/* If user attached a file, show a small preview */}
   481	              {msg.originalFile && (
   482	                <div className={`flex items-center gap-3 mb-3 p-3 rounded-xl ${msg.role === 'user' ? 'bg-indigo-700/50' : 'bg-slate-700/50'}`}>
   483	                  {msg.originalFile.type.startsWith('image/') ? (
   484	                    <ImageIcon size={18} className="shrink-0" />
   485	                  ) : (
   486	                    <FileText size={18} className="shrink-0" />
   487	                  )}
   488	                  <span className="text-xs truncate max-w-[200px] font-medium">{msg.originalFile.name}</span>
   489	                </div>
   490	              )}
   491	              {msg.parts.map((part: any, i: number) => {
   492	                if (part.text) {
   493	                  return <div key={i} className="whitespace-pre-wrap text-sm">{part.text}</div>;
   494	                }
   495	                return null;
   496	              })}
   497	            </div>
   498	          </div>
   499	        ))}
   500	        {isLoading && (
   501	          <div className="flex justify-start">
   502	             <div className="bg-slate-50 dark:bg-slate-800/40 border border-slate-200 dark:border-slate-700/30 rounded-2xl rounded-tl-sm px-5 py-4 flex items-center gap-3 text-slate-400">
   503	               <Loader2 size={16} className="animate-spin" />
   504	               <span className="text-sm">Processing...</span>
   505	             </div>
   506	          </div>
   507	        )}
   508	        <div ref={messagesEndRef} />
   509	      </div>
   510	
   511	      {/* Input Area */}
   512	      <div className="p-4 border-t border-slate-800 bg-slate-900/50">
   513	        
   514	        {/* Selected File Preview */}
   515	        {selectedFile && (
   516	          <div className="flex items-center gap-3 mb-3 px-3 py-2 bg-slate-50 dark:bg-slate-800/40 border border-slate-200 dark:border-slate-700/30 rounded-xl max-w-fit">
   517	            {selectedFile.type.startsWith('image/') ? (
   518	               // eslint-disable-next-line @next/next/no-img-element
   519	               <img src={`data:${selectedFile.type};base64,${selectedFile.base64}`} alt="preview" className="w-10 h-10 object-cover rounded-lg" />
   520	            ) : (
   521	               <div className="w-10 h-10 bg-slate-700/50 rounded-lg flex items-center justify-center text-slate-400">
   522	                  <FileText size={18} />
   523	               </div>
   524	            )}
   525	            <div className="flex flex-col">
   526	              <span className="text-xs font-bold text-slate-800 dark:text-slate-200 max-w-[200px] truncate">{selectedFile.name}</span>
   527	              <span className="text-[10px] text-slate-500 dark:text-slate-500 font-medium uppercase tracking-wider mt-0.5">Attached asset</span>
   528	            </div>
   529	            <button onClick={removeFile} className="ml-3 text-slate-500 dark:text-slate-500 hover:text-rose-400 transition-colors">
   530	              <svg width="14" height="14" viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2"><path d="M18 6L6 18M6 6l12 12"/></svg>
   531	            </button>
   532	          </div>
   533	        )}
   534	
   535	        <div className="flex items-end gap-2 relative">
   536	          <button 
   537	            onClick={() => fileInputRef.current?.click()}
   538	            className="p-3 text-slate-500 dark:text-slate-500 hover:text-indigo-400 hover:bg-indigo-500/10 rounded-xl transition-colors shrink-0 border border-slate-200 dark:border-slate-700/30 bg-slate-50 dark:bg-slate-800/40"
   539	            title="Attach Document"
   540	          >
   541	            <Paperclip size={20} />
   542	          </button>
   543	          <input 
   544	            type="file" 
   545	            ref={fileInputRef}
   546	            onChange={handleFileChange}
   547	            className="hidden"
   548	            accept="image/*,application/pdf,text/plain"
   549	          />
   550	          
   551	          <div className="flex-1 relative">
   552	            <textarea
   553	              value={input}
   554	              onChange={(e) => setInput(e.target.value)}
   555	              onKeyDown={(e) => {
   556	                if (e.key === 'Enter' && !e.shiftKey) {
   557	                  e.preventDefault();
   558	                  handleSend();
   559	                }
   560	              }}
   561	              placeholder="Message Hermes, or drag and drop a document..."
   562	              className="w-full max-h-32 min-h-[46px] p-3 py-3 rounded-xl border border-slate-200 dark:border-slate-700/30 bg-slate-50 dark:bg-slate-800/40 text-slate-800 dark:text-slate-200 focus:outline-none focus:ring-2 focus:ring-indigo-500/50 resize-none text-sm placeholder:text-slate-500 dark:text-slate-500"
   563	              rows={1}
   564	            />
   565	          </div>
   566	          
   567	          <button 
   568	            onClick={handleSend}
   569	            disabled={(!input.trim() && !selectedFile) || isLoading}
   570	            className="p-3 bg-indigo-600 hover:bg-indigo-500 disabled:opacity-50 disabled:cursor-not-allowed text-white rounded-xl transition-colors shrink-0 shadow-lg shadow-indigo-500/20"
   571	          >
   572	            <Send size={20} />
   573	          </button>
   574	        </div>
   575	        <div className="text-center mt-3 text-[10px] text-slate-500 dark:text-slate-500 uppercase tracking-widest font-semibold">
   576	           Verify important information. Supports Images, PDFs, and Text files.
   577	        </div>
   578	      </div>
   579	    </div>
   580	  );
   581	}
